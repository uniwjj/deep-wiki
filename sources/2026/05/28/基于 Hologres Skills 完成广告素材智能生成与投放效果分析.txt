广告素材生成的挑战
以游戏行业为例，在广告素材生成流程中面临如下多个结构性瓶颈：

创意产能瓶颈-做的慢

效果验证瓶颈-测得慢

全链路闭环瓶颈-转不动

• 传统素材生成靠堆人，创意风格不一

• 以来人工经验判断素材潜力，缺乏创新

• 素材生产与效果数据割裂，缺乏有

致，创意落地偏差大，沟通损耗严

筛选机制，凭感觉堵爆款，试错成本高

重，人工扩展边界成本高
• 流水线成本高、周期长，素材迭代速
度赶不上投放节奏

• 测试素材池小，全链路周期10-15天，
ROI难以提升，错过红利期

效的下一轮创意指导
• 投放优化依赖人工，全链路难以规
模化，自动化迭代。

Shot 3 (6-8s): 高光时刻 — 特效/大招/胜利瞬间
Shot 4 (8-10s): 结尾 — Slogan + 下载引导

ai_gen(
'wan',

json_build_object(

-- 分镜脚本内容
'reference_urls',
-- 游戏素材图片
material_list,
'parameters', json_build_object(
'size',
'1280*720', -- 720P 分辨率
'duration', 10,
-- 10秒视频
'shot_type', 'multi',
-- 多分镜模式
'audio', false,
-- 无音频
'watermark', true
-- 加水印
),
'output_url',
-- 输出到 OSS
'oss://ai-demo-dataset/xxx/output/'
)::text,
to_file(
'oss://ai-demo-dataset/xxx/man.png',
'oss-cn-hangzhou-internal.aliyuncs.com',
'acs:ram::***role'
)
), script
FROM story_script;

Step 4：素材审核以及投放效果分析
素材自动审核

广告投放实时分析

SELECT
material_id,
oss_path,
ai_gen(
'qwen_image_2',
json_build_object(
'prompt', format(
'审核素材%s：评估合规性/平台规范/品牌质量，返回JSON格式{"score":0100,"status":"approved/rejected","issues":[]}',
oss_path
),
'parameters', json_build_object('temperature', 0.1)
)::TEXT,
NULL
) AS review_result
FROM creative_materials
WHERE status = 'pending';
